import {{'{'}}TableVirtualScrollModule} from 'ng-table-virtual-scroll';

Directives

TableItemSizeDirective

A virtual scroll strategy for mat-table that supports fixed-size items.

Selector: cdk-virtual-scroll-viewport[tvsItemSize]

Properties
Name Description
@Input()
tvsItemSize: number; 		The size of the row in the table (in pixels). Defaults to 48.
@Input()
headerHeight: number; 		The size of the header row in the table (in pixels). Defaults to 56.
@Input()
footerHeight: number; 		The size of the footer row in the table (in pixels). Defaults to 48.
@Input()
headerEnabled: boolean; 		Is the header in the table (will affect on calculated table height) Defaults to true.
@Input()
footerEnabled: boolean; 		Is the footer in the table (will affect on calculated table height) Defaults to false.
@Input()
bufferMultiplier: number; 		The 'bufferMultiplier * visibleRowsCount' number of rows will be rendered before and after visible part of the table. Defaults to 0.7.
